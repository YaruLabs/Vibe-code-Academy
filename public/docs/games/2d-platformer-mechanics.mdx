#### Coyote Time
```javascript
// Allow jumping shortly after leaving platform
const COYOTE_TIME = 0.1; // seconds
let coyoteTimer = 0;

function update(deltaTime) {
  if (!isGrounded) {
    coyoteTimer += deltaTime;
  } else {
    coyoteTimer = 0;
  }

  canJump = isGrounded || coyoteTimer < COYOTE_TIME;
}
```
